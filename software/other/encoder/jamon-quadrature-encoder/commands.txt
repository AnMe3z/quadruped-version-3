PICO SDK SETUP

install pico sdk locally inside /home/{USER}/

export PICO_SDK_PATH=/home/andrey-shefa/pico/pico-sdk/

sudo apt get minicom

HOW TO CREATE A NEW PROJECT

- setup

mkdir leg-controller
cd leg-controller

code .

create main.c and CMakeLists.txt files inside

cp /home/{USER}/pico/pico-sdk/external/pico_sdk_import.cmake .

cp /home/{USER}/pico/pico-examples/pico_w/wifi/lwipopts_examples_common.h lwipopts.h

echo "$PICO_SDK_PATH"

- building

mkdir build

cd build/

cmake .. -DPICO_BOARD=pico_w ..

make

- uploading

cp leg_controller_project.uf2 /media/andrey-shefa/RPI-RP2/

- open serial minitor

dmesg | grep tty

sudo minicom -b 115200 -o -D /dev/ttyACM0
